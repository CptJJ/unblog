import { html2md } from "@/lib/crawler/html2md";
import { client } from "@/trigger";
import { invokeTrigger } from "@trigger.dev/sdk";
import { z } from "zod";

// Your first job
// This Job will be triggered by an event, log a joke to the console, and then wait 5 seconds before logging the punchline.
export const parseUrlJob = client.defineJob({
  // This is the unique identifier for your Job, it must be unique across all Jobs in your project.
  id: "crawl-page",
  name: "Crawls a single page, getting MD from its content and its SEO metadata",
  version: "0.0.1",
  // This is triggered by an event using eventTrigger. You can also trigger Jobs with webhooks, on schedules, and more: https://trigger.dev/docs/documentation/concepts/triggers/introduction
  trigger: invokeTrigger(),

  run: async (payload, io, ctx) => {
    const siteMd = await io.runTask("get-page-md", async () => {
      return await html2md(payload.url);
    });

    await io.logger.info(siteMd);
  },
});
